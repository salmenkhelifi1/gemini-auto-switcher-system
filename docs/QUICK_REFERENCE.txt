💻 HOW TO USE
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  BASIC USAGE:
  ────────────────────────────────────────────────────────────────────
  curl -X POST "https://n8n.iwilltravelto.com/webhook/apify-scrape" \
    -H "Content-Type: application/json" \
    -d '{"searchQuery": "restaurants in Paris"}'

  WITH CUSTOM MAX PLACES:
  ────────────────────────────────────────────────────────────────────
  curl -X POST "https://n8n.iwilltravelto.com/webhook/apify-scrape" \
    -H "Content-Type: application/json" \
    -d '{"searchQuery": "hotels in London", "maxPlaces": 50}'

  PARAMETERS:
  ────────────────────────────────────────────────────────────────────
    searchQuery   (required)  - What to search
    maxPlaces     (optional)  - How many places (default: 20)
    language      (optional)  - Language code (default: "en")

📊 DATA SAVED TO GOOGLE SHEETS
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  Sheet: ScrapedPlaces
  
  Columns:
  ├─ name              - Place name
  ├─ address           - Full address
  ├─ rating            - Star rating (1-5)
  ├─ reviews           - Number of reviews
  ├─ phone             - Phone number
  ├─ website           - Website URL
  ├─ category          - Category name
  ├─ priceLevel        - Price level ($-$$$$)
  ├─ latitude          - GPS latitude
  ├─ longitude         - GPS longitude
  ├─ googleMapsUrl     - Google Maps link
  ├─ imageUrl          - Image URL
  ├─ openingHours      - Opening hours
  ├─ isOpen            - Is currently open (true/false)
  └─ scrapedAt         - Timestamp
